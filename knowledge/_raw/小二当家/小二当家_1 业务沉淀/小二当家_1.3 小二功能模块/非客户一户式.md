# 基本信息模块

接口：/company/getCompanyDetailByTaxNo

```java
-- 查询非客户的基本信息
select binfo.id, binfo.customer_name, binfo.tax_no ,binfo.social_credit_code ,binfo.biz_area_code ,binfo.registration_capital ,binfo.industry_category ,binfo.main_business_scope ,
binfo.vat_taxpayer_type ,binfo.registration_type ,binfo.registration_address ,
binfo.establish_time ,bussinfo.from_time ,bussinfo.to_time ,bussinfo.reg_status ,
taxinfo.main_tax_bureau_code ,taxinfo.main_tax_bureau_branch_code ,taxinfo.status ,
contactinfo.legal_person_name ,contactinfo.legal_person_mobile 
from collect_company_base_info binfo
left join collect_company_tax_info taxinfo  on binfo.id = taxinfo.company_id 
left join collect_company_business_info bussinfo  on binfo.id = bussinfo.company_id 
left join collect_company_contact_info contactinfo on binfo.id =contactinfo.company_id 
where 1=1
and binfo.id = '174776'
;
```
