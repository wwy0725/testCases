展示数据范围：当前客户经理的责任个代，市场划分md\_customer\_divide\_info\_institution表中客户经理为当前登录者的所有中介

筛选条件及表头可配，配置在itcrm\_table\_configuration

数据由数据组推送到 ads\_itcrm\_agent\_company\_relation

```sql
select
        a.manager_divide_time divideTime,
        b.county_code countyCode,
        b.county_name countyName,
        b.person_name fullName,
        b.person_id personId,
        b.agent_id institutionId,
        b.agent_name agentName,
        b.mobile,
        case when b.wechat_customer_manager like '%${wechatCustomerManager}%' then 'Y' else 'N' end addMyWorkWeChat,
        case when b.wechat_customer_manager like '%${wechatCustomerManager}%' then '是' else '否' end addMyWorkWeChatName,
        b.wechat_customer_manager wechatCustomerManager,
        b.member_grade memberGrade,
        b.sale_amount saleAmount,
        b.pre_sale_balance preSaleBalance,
        b.total_cnt totalCnt,
        b.vip_cnt vipCnt,
        b.nonvip_cnt nonvipCnt,
        b.nonvip_general_cnt nonvipGeneralCnt,
        b.nonvip_small_scale_cnt nonvipSmallScaleCnt,
        b.rights_cnt rightsCnt,
        b.service_cnt serviceCnt,
        b.invoice_cnt invoiceCnt,
        b.finance_cnt financeCnt,
        b.tax_cnt taxCnt,
        b.risk_cnt riskCnt,
        b.guard_cnt guardCnt,
        b.expire_cnt expireCnt,
        e.followNumber ,
        e.create_time lastFollowTime,
        e.text_content followRecord,
        count(DISTINCT b.company_pc_id) companyNumber,
        concat(if(agent_star_level = '0','非',agent_star_level), if(agent_level = 'E','-',agent_level)) starLevel
        from market_divide.md_customer_divide_info_institution a join ads_itcrm_agent_company_relation b on
        a.customer_id = b.agent_id
        left join (select c.*,d.followNumber,d.institution_id
        from itcrm_follow_record_info c join (select t.institution_id,max(t.follow_record_id) maxid,count(distinct
        t.follow_record_id, t.institution_id) followNumber
        from itcrm_follow_record_institution_relation t group by t.institution_id) d on c.id = d.maxid ) e on b.agent_id
        = e.institution_id
        where a.customer_manager = #{customerManager}
        <if test="isVip != null and isVip != ''">
            and b.is_vip = #{isVip}
        </if>
        <if test="cityCode != null and cityCode != ''">
            and b.city_code = #{cityCode}
        </if>
        <if test="countyCode != null and countyCode != ''">
            and b.county_code = #{countyCode}
        </if>
        <if test="taxRegisterDateFrom != null and taxRegisterDateFrom != ''">
            and b.tax_register_date >= #{taxRegisterDateFrom}
        </if>
        <if test="taxRegisterDateTo != null and taxRegisterDateTo != ''">
            <![CDATA[ and b.tax_register_date <= #{taxRegisterDateTo} ]]>
        </if>
        <if test="lastActiveDateFrom != null and lastActiveDateFrom != ''">
            <![CDATA[ and b.last_active_date < #{lastActiveDateFrom} ]]>
        </if>
        <if test="memberGrade != null and memberGrade != ''">
            and b.member_grade = #{memberGrade}
        </if>
        <if test="addMyWorkWeChat != null and addMyWorkWeChat == 'Y'.toString()">
            and b.wechat_customer_manager like concat('%', #{wechatCustomerManager}, '%')
        </if>
        <if test="addMyWorkWeChat != null and addMyWorkWeChat == 'N'.toString()">
            and (b.wechat_customer_manager not like concat('%', #{wechatCustomerManager}, '%') or
            b.wechat_customer_manager is null)
        </if>
        <if test="lastFollowTimeFrom != null and lastFollowTimeFrom != ''">
            and e.create_time >= #{lastFollowTimeFrom}
        </if>
        <if test="lastFollowTimeTo != null and lastFollowTimeTo != ''">
            <![CDATA[ and e.create_time <= #{lastFollowTimeTo} ]]>
        </if>
        <if test="productExpireDateTo != null and productExpireDateTo != ''">
            <![CDATA[ and product_expire_date <= #{productExpireDateTo} ]]>
        </if>
        <if test="productExpireDateFrom != null and productExpireDateFrom != ''">
            and product_expire_date >= #{productExpireDateFrom}
        </if>
        <if test="vatTaxpayerType != null and vatTaxpayerType != ''">
            and b.vat_taxpayer_type = #{vatTaxpayerType}
        </if>
        <if test="agentLevel != null and agentLevel != ''">
            and b.agent_level = #{agentLevel}
        </if>
        <if test="agentStar != null and agentStar != ''">
            and b.agent_star_level = #{agentStar}
        </if>
        <if test="agentCustomerNumFrom != null and agentCustomerNumFrom != ''">
            and b.total_cnt >= #{agentCustomerNumFrom}
        </if>
        <if test="agentCustomerNumTo != null and agentCustomerNumTo != ''">
            <![CDATA[ and b.total_cnt <= #{agentCustomerNumTo} ]]>
        </if>
        <if test="agentVipCustomerNumFrom != null and agentVipCustomerNumFrom != ''">
            <![CDATA[ and b.vip_cnt >= #{agentVipCustomerNumFrom} ]]>
        </if>
        <if test="agentVipCustomerNumTo != null and agentVipCustomerNumTo != ''">
            <![CDATA[ and b.vip_cnt <= #{agentVipCustomerNumTo} ]]>
        </if>
        <if test="agentNonVipCustomerNumFrom != null and agentNonVipCustomerNumFrom != ''">
            <![CDATA[ and b.nonvip_cnt >= #{agentNonVipCustomerNumFrom} ]]>
        </if>
        <if test="agentNonVipCustomerNumTo != null and agentNonVipCustomerNumTo != ''">
            <![CDATA[ and b.nonvip_cnt <= #{agentNonVipCustomerNumTo} ]]>
        </if>
        <if test="institutionName != null and institutionName != ''">
            and b.agent_name like concat('%', #{institutionName}, '%')
        </if>
        group by b.agent_id
        order by
        <if test="orderBy != null and orderBy != '' and orderBy != 'starLevel'.toString()">
            ${orderBy}
            ${orderByType},
            b.id
        </if>
        <if test="orderBy != null and orderBy != '' and orderBy == 'starLevel'.toString()">
            agent_star_level ${orderByTypeStarLevel},
            agent_level ${orderByType},
            b.id
        </if>
```
