```java
-- 合同列表
SELECT HT_ID 合同id, QSRQ 起始时间, DQRQ 到期时间, MC 分期, DD_JE 合同金额, DD_ZTSL 总账套数, YBRZTSL 一般人,

FROM YYFX.ADS_INST_CONTRACT_LIST;

-- 合同明细
SELECT HT_ID 合同id, QDSJ 合同签订时间, ZJGDM 机构代码, QDJG 机构名称, JGID1 关联代码1, JGID2 关联代码2, LXR 联系人, LXDH 联系方式,

PJDJ 一般人客单价, XGMZTSL 小规模账套数, XPJDJ 小规模客单价, GTHZTSL 个体户账套数, GTHKDJ 个体户客单价, ZSXHDKSL 约定新户数, HJML 户均毛利, 
YI_SKRQ 第一次收款日期, ER_SKRQ 第二次收款日期 , SA_SKRQ 第三次收款日期, SI_SKRQ 第四次收款日期, WU_SKRQ 第五次收款日期, YI_SKJED 第一次收款金额, ER_SKJED 第二期金额, 
SA_SKJED 第三期金额, SI_SKJED 第四期金额, WU_SKJED 第五期金额, CSZJZXSJH1  授权手机号1, CSZJZXSJH2  授权手机号2, CSZJZXSJH3  授权手机号3, 
CSZJZXSJH4 授权手机号4 , CSZJZXSJH5  授权手机号5, HTBZ 合同备注, YGBZ 员工备注, QYHS 签约户数, YDGTS 约定个体数, FKFS 付款方式, KX 可选, YX 已选
FROM YYFX.ADS_INST_CONTRACT_DETAIL
WHERE  HT_ID ='ZFHBLF20210035';

-- 回款信息
SELECT HT_ID 合同id, SKRQ 回款时间, HKFS 回款时间, FLOW_ID 回款单号, SKJE 回款金额
FROM YYFX.ADS_INST_CONTRACT_REPAYMENT;

--合同照片
SELECT HT_ID 合同id, HTZP 合同照片, YS 友商, FWQD 服务清单, ZZHTZP 中止合同照片, LDTP 领导特批
FROM YYFX.ADS_INST_CONTRACT_PHOTO;

--服务清单
SELECT HT_ID 合同id, MC 服务项目, DXNY 对象时间
FROM YYFX.ADS_INST_CONTRACT_SERVICE;
```
